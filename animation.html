    html, body {

    #main {

      width: 100vmin;

      filter: blur(20px) contrast(500%);

    }

    #r1, #r2, #r3, #r4 {

      width: 30vmin; height: 30vmin;

      position: absolute;

    }

    #r1 { top: 10vmin; left: 10vmin; }

    #r2 { top: 10vmin; right: 10vmin; transform: rotateY(180deg); }

    #r3 { bottom: 10vmin; left: 10vmin; transform: rotateX(180deg); }

    #r4 { bottom: 10vmin; right: 10vmin; transform: rotateX(180deg) rotateY(180deg); }

    .circle {

      background: #000;

      width: 5vmin; height: 5vmin;

      top: 14vmin; left: 14vmin;

      border: 0.6vmin solid #fff;

      border-radius: 50%;

      opacity: 0;

      position: absolute;

      animation: rotate 1s linear infinite;

      transition: 0.5s linear;

<body>

  <div id="main">

    <div id="r1"></div>

    <div id="r2"></div>

    <div id="r3"></div>

    <div id="r4"></div>

  </div>


  <script>

    var nbr = 4;

    for (let i = 0; i < nbr; i++) {

      ['r1','r2','r3','r4'].forEach(id => {

        let div = document.createElement('div');

        div.className = 'circle';

        document.getElementById(id).appendChild(div);

        setTimeout(() => div.style.opacity = '0.5', 500);

      });

    }


    setInterval(() => {

      document.querySelectorAll('.circle').forEach(c => {

        c.style.background = `rgb(${randomRGB()})`;

        let m = randomMargin(), s = randomSize(), bd = randomRadius();

        c.style.margin = `${m}px 0 0 ${m}px`;

        c.style.width = c.style.height = `${s}px`;

        c.style.borderRadius = `${bd}px`;

      });

    }, 500);


    setInterval(() => {

      document.body.style.background = `rgb(${randomRGB()})`;

    }, 2000);


    function randomRGB(){

      return [0,1,2].map(_=> Math.floor(Math.random()*255)).join(',');

    }

    function randomMargin(){ return Math.floor(Math.random()*100); }

    function randomSize(){ return Math.floor(Math.random()*90 +10); }

    function randomRadius(){ return Math.floor(Math.random()*100); }

  </script>
